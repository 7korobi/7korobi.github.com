一列目
``` ruby
module A
  def a
  end
  A = 1
end

class B
  include A
end

class C
  extend A
end
```

二列目
``` ruby
module A
  def a
  end
  A = 1
  module_function :a
end

class B
  include A
end

class C
  extend A
end
```

三列目
``` ruby
module A
  def self.a
  end
  A = 1
end

class B
  include A
end

class C
  extend A
end
```
